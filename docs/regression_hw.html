model_distance <- function( interscept, slope, X, Y ) {
  yhat <- interscept + slope * X
  diff <- Y - yhat
  return( sqrt( mean( diff ^ 2 ) ) )
}


grid <- expand.grid( beta0 = seq(15, 
                                 20, 
                                 length = 25),
                     beta1 = seq(2, 
                                 3, 
                                 length = 25))
grid$dist <- NA
for( i in 1:nrow(grid) ) {
  grid$dist[i] <- model_distance( grid$beta0[i],
                                  grid$beta1[i],
                                  df$x,
                                  df$y )
}

ggplot( grid, aes(x = beta0, 
                  y = beta1,
                  color = -dist)) + 
  geom_point( data = filter( grid, 
                             rank(dist) <= 10), 
              color = "red",
              size = 4) +
  geom_point()
model <- lm(beta1 ~ beta0, data = grid)

slope <- coef(model)[[2]]

print(slope)
